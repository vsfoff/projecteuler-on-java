class Fib{
    public static int fibb(int n){
        if(n==0){ return 0; }
        if(n==1||n==2){ return 1; }
        return fibb(n-2)+fibb(n-1);
    }
}
public class Main extends Fib{

    public static void main(String[] args){
        int max=39;
        int sum=0;
        for(int i=0;i<max;i++){
            if(fibb(i)%2==0){
                sum=i*i;
            }
        }System.out.println(sum);
    }
}
